20 Exemplos de C++ Moderno (comentados)

1. Leitura de arquivo TXT linha a linha

// Lê cada linha de um arquivo .txt
#include <fstream>
#include <iostream>
#include <string>

void lerTXT(const std::string& caminho) {
    std::ifstream arquivo(caminho);
    std::string linha;
    while (std::getline(arquivo, linha)) {
        std::cout << linha << '
';
    }
}

2. Leitura de CSV simples (separador vírgula)

// Lê campos separados por vírgula
#include <sstream>
#include <vector>

std::vector<std::vector<std::string>> lerCSV(const std::string& caminho) {
    std::ifstream arquivo(caminho);
    std::vector<std::vector<std::string>> dados;
    std::string linha;
    while (std::getline(arquivo, linha)) {
        std::stringstream ss(linha);
        std::string campo;
        std::vector<std::string> linhaDados;
        while (std::getline(ss, campo, ',')) {
            linhaDados.push_back(campo);
        }
        dados.push_back(linhaDados);
    }
    return dados;
}

3. Leitura de TXT com delimitador personalizado

// Usa '|' como delimitador
#include <fstream>
#include <string>
#include <sstream>

void lerDelimitado(const std::string& caminho, char delim = '|') {
    std::ifstream arq(caminho);
    std::string linha;
    while (std::getline(arq, linha)) {
        std::stringstream ss(linha);
        std::string token;
        while (std::getline(ss, token, delim)) {
            std::cout << token << ' ';
        }
        std::cout << '
';
    }
}

4. Ler arquivo inteiro para uma string (moderno)

// Usa istreambuf_iterator para ler tudo de uma vez
#include <fstream>
#include <string>
#include <iterator>

std::string lerArquivoInteiro(const std::string& caminho) {
    std::ifstream arq(caminho);
    return {std::istreambuf_iterator<char>(arq),
            std::istreambuf_iterator<char>()};
}

5. Bitwise: verificar se número é potência de 2

// n > 0 e (n & (n-1)) == 0
bool ehPotenciaDe2(int n) {
    return n > 0 && (n & (n - 1)) == 0;
}

8. Bitwise: encontrar elemento único em array (XOR)

// Todos os outros aparecem duas vezes
int unico(const std::vector<int>& v) {
    int res = 0;
    for (int x : v) res ^= x;
    return res;
}

9. Bitwise: definir/limpar/alternar um bit

// Operações em um byte
uint8_t setBit(uint8_t valor, int bit) { return valor | (1 << bit); }
uint8_t clearBit(uint8_t valor, int bit) { return valor & ~(1 << bit); }
uint8_t toggleBit(uint8_t valor, int bit) { return valor ^ (1 << bit); }
bool testBit(uint8_t valor, int bit) { return valor & (1 << bit); }

13. constexpr função – avaliada em tempo de compilação

constexpr int fatorial(int n) {
    return (n <= 1) ? 1 : n * fatorial(n - 1);
}
// static_assert(fatorial(5) == 120);

14. std::optional – valor opcional

#include <optional>

std::optional<double> dividir(double a, double b) {
    if (b == 0) return std::nullopt;
    return a / b;
}

20. std::filesystem – ler diretório e arquivos (C++17)

#include <filesystem>
namespace fs = std::filesystem;

void listarArquivos(const fs::path& dir) {
    for (const auto& entry : fs::directory_iterator(dir)) {
        if (entry.is_regular_file()) {
            std::cout << entry.path().filename() << '
';
        }
    }
}
