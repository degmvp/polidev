01 • Permission Flags

Controle de permissões de usuário

Simula permissões de acesso em sistemas bancários usando bits para READ, WRITE e EXEC.

#include <iostream>
#include <cstdint>

enum Permission {
    READ  = 0,
    WRITE = 1,
    EXEC  = 2
};

uint32_t setBit(uint32_t value, int bit) {
    return value | (1u << bit);
}

bool isBitSet(uint32_t value, int bit) {
    return (value & (1u << bit)) != 0;
}

uint32_t grantPermission(uint32_t userFlags, Permission p) {
    return setBit(userFlags, p);
}

bool hasPermission(uint32_t userFlags, Permission p) {
    return isBitSet(userFlags, p);
}

int main() {
    uint32_t user = 0;
    user = grantPermission(user, READ);
    user = grantPermission(user, WRITE);
    std::cout << "Permissao de leitura? " << hasPermission(user, READ) << "\n";
    std::cout << "Permissao de execucao? " << hasPermission(user, EXEC) << "\n";
    return 0;
}
02 • Bit Mask

Máscara para extrair código de agência

Extrai uma parte específica de um identificador binário empacotado.

#include <iostream>
#include <cstdint>

uint32_t maskBits(uint32_t value, int start, int length) {
    uint32_t mask = ((1u << length) - 1u) << start;
    return (value & mask) >> start;
}

uint32_t extractAgencyCode(uint32_t accountData) {
    return maskBits(accountData, 8, 8);
}

int main() {
    uint32_t accountPacked = 0b1010101100110010;
    std::cout << "Agencia: " << extractAgencyCode(accountPacked) << "\n";
    return 0;
}
03 • Transaction Flags

Compressão de flags de transação

Compacta múltiplos estados booleanos em um único inteiro.

#include <iostream>
#include <cstdint>

uint32_t setBit(uint32_t value, int bit) {
    return value | (1u << bit);
}

uint32_t compressFlags(bool approved, bool fraud, bool priority) {
    uint32_t flags = 0;
    if (approved) flags = setBit(flags, 0);
    if (fraud)    flags = setBit(flags, 1);
    if (priority) flags = setBit(flags, 2);
    return flags;
}

int main() {
    uint32_t txFlags = compressFlags(true, false, true);
    std::cout << "Flags da transacao: " << txFlags << "\n";
    return 0;
}
04 • Rotation + XOR

Rotação para criptografia simples

Demonstra criptografia leve combinando rotação de bits e XOR.

#include <iostream>
#include <cstdint>

uint32_t rotateLeft(uint32_t value, int shift) {
    return (value << shift) | (value >> (32 - shift));
}

uint32_t rotateRight(uint32_t value, int shift) {
    return (value >> shift) | (value << (32 - shift));
}

uint32_t simpleEncrypt(uint32_t value) {
    return rotateLeft(value, 5) ^ 0xA5A5A5A5;
}

uint32_t simpleDecrypt(uint32_t value) {
    return rotateRight(value ^ 0xA5A5A5A5, 5);
}

int main() {
    uint32_t data = 123456;
    auto enc = simpleEncrypt(data);
    auto dec = simpleDecrypt(enc);
    std::cout << "Original: " << data << " | Decriptado: " << dec << "\n";
    return 0;
}
05 • Integrity Check

Verificação de integridade

Valida uma condição estrutural do pacote usando menor bit ativo.

#include <iostream>
#include <cstdint>

uint32_t lowestBit(uint32_t value) {
    return value & -value;
}

bool isPowerOfTwo(uint32_t value) {
    return value != 0 && (value & (value - 1)) == 0;
}

bool checkIntegrity(uint32_t value) {
    return isPowerOfTwo(lowestBit(value));
}

int main() {
    uint32_t packet = 0b1001000;
    std::cout << "Integridade ok? " << checkIntegrity(packet) << "\n";
    return 0;
}
06 • Toggle Flag

Controle de status de transações

Alterna rapidamente o estado de fraude em uma transação.

#include <iostream>
#include <cstdint>

uint32_t toggleBit(uint32_t value, int bit) {
    return value ^ (1u << bit);
}

void toggleFraudFlag(uint32_t& txFlags) {
    txFlags = toggleBit(txFlags, 1);
}

int main() {
    uint32_t txFlags = 0b001;
    toggleFraudFlag(txFlags);
    std::cout << "Flags apos toggle: " << txFlags << "\n";
    return 0;
}
07 • Popcount

Contagem de operações ativas

Conta quantos processos estão ativos em um sistema usando flags.

#include <iostream>
#include <cstdint>

int countBits(uint32_t value) {
    int count = 0;
    while (value) {
        value &= (value - 1);
        ++count;
    }
    return count;
}

int activeOperations(uint32_t opsFlags) {
    return countBits(opsFlags);
}

int main() {
    uint32_t ops = 0b101101;
    std::cout << "Operacoes ativas: " << activeOperations(ops) << "\n";
    return 0;
}
08 • Power of Two

Validação de blocos de memória

Garante que blocos de memória respeitem tamanhos potência de 2.

#include <iostream>

bool isPowerOfTwo(uint32_t value) {

09 • Lowest Active Bit

Extração do menor bit ativo para auditoria

Identifica a menor flag ativa em um conjunto de estados.

#include <iostream>
#include <cstdint>

uint32_t lowestBit(uint32_t value) {
    return value & -value;
}

void auditLowestFlag(uint32_t flags) {
    uint32_t lsb = lowestBit(flags);
    std::cout << "Menor flag ativa: " << lsb << "\n";
}

int main() {
    auditLowestFlag(0b101100);
    return 0;
}
10 • Rotate Hash

Rotação para gerar hash simples

Combina rotações opostas para produzir uma assinatura binária simples.

#include <iostream>
#include <cstdint>

uint32_t rotateLeft(uint32_t value, int shift) {
    return (value << shift) | (value >> (32 - shift));
}

uint32_t rotateRight(uint32_t value, int shift) {
    return (value >> shift) | (value << (32 - shift));
}

uint32_t generateHash(uint32_t key) {
    return rotateLeft(key, 7) ^ rotateRight(key, 3);
}

int main() {
    uint32_t key = 12345;
    std::cout << "Hash: " << generateHash(key) << "\n";
    return 0;
}

O objetivo desta série é mostrar como esse controle pode ser aplicado com clareza, contexto e disciplina de engenharia.
