01. Verificar se uma data é fim de semana

CREATE FUNCTION dbo.IsWeekend(@Date DATE)
RETURNS BIT
AS
BEGIN
    -- Retorna 1 se a data for sábado ou domingo, caso contrário, 0
    RETURN CASE WHEN DATEPART(WEEKDAY, @Date) IN (7, 1) THEN 1 ELSE 0 END
END

Útil para filtrar ou marcar registros que caem em finais de semana, ajudando em análises de produtividade ou vendas.

02. Obter o último dia do mês

CREATE FUNCTION dbo.LastDayOfMonth(@Date DATE)
RETURNS DATE
AS
BEGIN
    -- Retorna o último dia do mês da data fornecida
    RETURN DATEADD(DAY, -DAY(DATEADD(MONTH, 1, @Date)), DATEADD(MONTH, 1, @Date))
END

Facilita cálculos de fechamento de períodos ou geração de relatórios mensais.

03. Calcular diferença em dias

CREATE FUNCTION dbo.DaysDifference(@StartDate DATE, @EndDate DATE)
RETURNS INT
AS
BEGIN
    -- Retorna a quantidade de dias entre duas datas
    RETURN DATEDIFF(DAY, @StartDate, @EndDate)
END

Essencial para métricas de tempo, como prazos ou períodos de validade.

04. Normalizar texto

CREATE FUNCTION dbo.NormalizeText(@Text VARCHAR(100))
RETURNS VARCHAR(100)
AS
BEGIN
    -- Remove espaços extras e converte para maiúsculas
    RETURN UPPER(LTRIM(RTRIM(@Text)))
END

Ajuda na padronização de dados antes de comparações ou inserções.

05. Verificar nulo ou vazio

CREATE FUNCTION dbo.IsNullOrEmpty(@Value VARCHAR(100))
RETURNS BIT
AS
BEGIN
    -- Retorna 1 se o valor for nulo ou vazio, caso contrário, 0
    RETURN CASE WHEN @Value IS NULL OR @Value = '' THEN 1 ELSE 0 END
END

Útil para validações rápidas em consultas complexas.

06. Gerar um GUID único

CREATE FUNCTION dbo.GenerateGUID()
RETURNS UNIQUEIDENTIFIER
AS
BEGIN
    -- Retorna um novo GUID
    RETURN NEWID()
END

Facilita a criação de chaves únicas para registros.

07. Percentual de crescimento

CREATE FUNCTION dbo.PercentageGrowth(@Previous DECIMAL(18,2), @Current DECIMAL(18,2))
RETURNS DECIMAL(5,2)
AS
BEGIN
    -- Calcula o percentual de crescimento
    RETURN CASE WHEN @Previous = 0 THEN NULL ELSE ((@Current - @Previous) / @Previous) * 100 END
END

Essencial para análises de desempenho ou vendas.

08. Buscar substring sem diferenciar maiúsculas/minúsculas

CREATE FUNCTION dbo.ContainsIgnoreCase(@Text VARCHAR(100), @SubString VARCHAR(50))
RETURNS BIT
AS
BEGIN
    -- Retorna 1 se @Text contém @SubString, ignorando maiúsculas/minúsculas
    RETURN CASE WHEN CHARINDEX(LOWER(@SubString), LOWER(@Text)) > 0 THEN 1 ELSE 0 END
END

Facilita buscas flexíveis em textos.

09. Calcular IMC

CREATE FUNCTION dbo.CalcIMC(@Weight DECIMAL(5,2), @Height DECIMAL(4,2))
RETURNS DECIMAL(4,2)
AS
BEGIN
    -- Calcula o IMC (peso em kg / altura em metros ao quadrado)
    RETURN @Weight / POWER(@Height, 2)
END

Útil em sistemas de saúde ou fitness.

10. Converter string em data com segurança

CREATE FUNCTION dbo.SafeConvertToDate(@DateString VARCHAR(10))
RETURNS DATE
AS
BEGIN
    -- Tenta converter a string para data, retornando NULL se inválido
    RETURN TRY_CAST(@DateString AS DATE)
END

Ajuda a evitar erros de conversão em consultas dinâmicas.
