✅ Decorator memorize (cache de chamadas)


class memorize(dict):
    def __init__(self, func):
        self.func = func

    def __call__(self, *args):
        return self[args]

    def __missing__(self, key):
        result = self[key] = self.func(*key)
        return result

✅ Decorator countcalls (contagem de chamadas)


class countcalls(object):
    "Decorator que conta quantas vezes a função foi chamada."

    __instances = {}

    def __init__(self, f):
        self.__f = f
        self.__numcalls = 0
        countcalls.__instances[f] = self

    def __call__(self, *args, **kwargs):
        self.__numcalls += 1
        return self.__f(*args, **kwargs)

    def count(self):
        "Retorna o número de chamadas da função decorada."
        return countcalls.__instances[self.__f].__numcalls

    @staticmethod
    def counts():
        "Retorna um dicionário com {função: número de chamadas}."
        return {
            f.__name__: countcalls.__instances[f].__numcalls
            for f in countcalls.__instances
        }

✅ Exemplo com função simples


@countcalls
def f():
    print('f called')

f()
f()

✅ Combinação de Decorators

A função foo(a, b) é decorada com @countcalls e @memorize. Ela realiza multiplicações com cache e contagem. 


@memorize
@countcalls
def foo(a, b):
    return a * b

print('Efetua 3 operações com a função foo(a, b): 2x4, 12x4, 12x12')
print('e armazena os resultados no decorador memorize (dicionário)')
print(foo(2, 4))
print(foo(12, 4))
print(foo(12, 12))
print(foo)
