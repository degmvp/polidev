✅ Gerador de identificadores

A função generator() retorna uma string aleatória formada por letras e dígitos hexadecimais. 


def generator():
    letter1 = random.choice(string.ascii_uppercase)
    letter2 = random.choice(string.ascii_uppercase)
    letter3 = random.choice(string.ascii_uppercase)
    letter4 = random.choice(string.ascii_lowercase)
    letter5 = random.choice(string.ascii_lowercase)
    number1 = random.choice(string.hexdigits)
    number2 = random.choice(string.hexdigits)
    number3 = random.choice(string.hexdigits)
    name = letter1 + letter2 + number1 + number2 + number3 + letter3 + letter4 + letter5
    return name

print(generator())

✅ Função Lógica com AND e NOT


def V_and_F(var_v, var_f):
    return var_v and not var_f

print(V_and_F(True, False))  # True
print(V_and_F(True, True))   # False

a = True
b = False
print(V_and_F(a, b))         # True
print(V_and_F(b, a))         # False
print(V_and_F(a, a))         # False

✅ Fibonacci com programação dinâmica (iterativa)

Implementação da sequência de Fibonacci sem recursão, utilizando iteração para maior eficiência. 


def fib_dyn(n):
    if n == 1 or n == 2:
        return 1
    f1 = 1
    f2 = 1
    for i in range(3, n + 1):
        f = f1 + f2
        f2 = f1
        f1 = f
    return f1

print(fib_dyn(7))  # Saída: 13
