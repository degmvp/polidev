RS-01

01 — Parallelismo com Rayon (data parallelism)

Demonstra paralelismo de dados com Rayon. Processa grandes coleções de forma eficiente em múltiplos núcleos.

// Cargo.toml: rayon = "1.10"
use rayon::prelude::*;

fn main() {
    let numbers: Vec<i32> = (0..1_000_000).collect();
    let sum: i32 = numbers.par_iter().map(|&x| x * x).sum();
    println!("Soma dos quadrados (paralelo): {}", sum);
}
RS-02

02 — Async/Await com Tokio (concorrência)

Exemplo prático de tarefas assíncronas com Tokio. Útil para I/O concorrente.

// Cargo.toml: tokio = { version = "1", features = ["full"] }
#[tokio::main]
async fn main() {
    let handles = (0..5).map(|i| {
        tokio::spawn(async move {
            println!("Tarefa {} iniciada", i);
            tokio::time::sleep(tokio::time::Duration::from_millis(100)).await;
            i * 10
        })
    }).collect::<Vec<_>>();

    for h in handles {
        println!("Resultado: {}", h.await.unwrap());
    }
}
RS-03

03 — Custom Iterator com State

Implementa um Iterator customizado. Demonstra zero-cost abstractions.

struct Fibonacci {
    curr: u64,
    next: u64,
}

impl Iterator for Fibonacci {
    type Item = u64;
    fn next(&mut self) -> Option<Self::Item> {
        let current = self.curr;
        self.curr = self.next;
        self.next += current;
        Some(current)
    }
}

fn main() {
    let fib: u64 = Fibonacci { curr: 0, next: 1 }.take(20).sum();
    println!("Soma dos primeiros 20 Fibonacci: {}", fib);
}
RS-04

04 — Trait Objects + Dyn Dispatch

Polimorfismo em tempo de execução com trait objects (vtable).

trait Drawable {
    fn draw(&self);
}

struct Circle;
struct Square;

impl Drawable for Circle { fn draw(&self) { println!("Desenhando círculo"); } }
impl Drawable for Square { fn draw(&self) { println!("Desenhando quadrado"); } }

fn main() {
    let shapes: Vec<Box<dyn Drawable>> = vec![Box::new(Circle), Box::new(Square)];
    for shape in shapes {
        shape.draw();
    }
}

05 — Unsafe + Raw Pointers (com segurança controlada)

Uso controlado de código unsafe para manipulação direta de memória. Cuidado: só para performance crítica.

fn main() {
    let mut data = vec![1, 2, 3, 4, 5];
    let ptr = data.as_mut_ptr();
    unsafe {
        *ptr.add(2) = 42;  // modifica índice 2
    }
    println!("Vetor modificado com unsafe: {:?}", data);
}

06 — Interior Mutability com RefCell

Mutabilidade interna em contexto single-threaded (borrow checker em runtime).

use std::cell::RefCell;

fn main() {
    let data = RefCell::new(vec![1, 2, 3]);
    {
        let mut borrowed = data.borrow_mut();
        borrowed.push(4);
    }
    println!("RefCell: {:?}", data.borrow());
}
RS-07

07 — Lock-free com Atomic (multi-thread)

Contador thread-safe sem mutex (melhor performance).

use std::sync::{Arc, atomic::{AtomicUsize, Ordering}};
use std::thread;

fn main() {
    let counter = Arc::new(AtomicUsize::new(0));
    let mut handles = vec![];

    for _ in 0..10 {
        let counter = counter.clone();
        handles.push(thread::spawn(move || {
            for _ in 0..1000 {
                counter.fetch_add(1, Ordering::Relaxed);
            }
        }));
    }

    for h in handles { h.join().unwrap(); }
    println!("Contador final: {}", counter.load(Ordering::SeqCst));
}
RS-08

08 — Macro Declarativa Avançada

Geração de código em tempo de compilação.

macro_rules! create_struct {
    ($name:ident, $($field:ident: $type:ty),*) => {
        struct $name {
            $($field: $type),*
        }
        impl $name {
            fn new($($field: $type),*) -> Self {
                Self { $($field),* }
            }
        }
    };
}

create_struct!(Person, name: String, age: u32);

fn main() {
    let p = Person::new("Alice".to_string(), 30);
    println!("{} tem {} anos", p.name, p.age);
}
RS-09

09 — Channels MPSC para Comunicação entre Threads

Comunicação thread-safe producer-consumer.

use std::sync::mpsc;
use std::thread;

fn main() {
    let (tx, rx) = mpsc::channel();
    thread::spawn(move || {
        for i in 0..5 {
            tx.send(i).unwrap();
            thread::sleep(std::time::Duration::from_millis(50));
        }
    });

    for received in rx {
        println!("Recebido: {}", received);
    }
}
RS-10

10 — Memory Mapping de Arquivo (mmap)

Acesso eficiente a arquivos grandes via memory mapping.

use std::fs::File;
use std::io::{self, Write};
use memmap2::MmapOptions; // Cargo.toml: memmap2 = "0.9"

fn main() -> io::Result<()> {
    let mut file = File::create("test.bin")?;
    file.write_all(b"Hello Rust mmap!")?;
    let file = File::open("test.bin")?;
    let mmap = unsafe { MmapOptions::new().map(&file)? };
    println!("Conteúdo mmap: {}", String::from_utf8_lossy(&mmap));
    Ok(())
}

11 — Error Handling com thiserror + anyhow (recomendado)

Tratamento de erros moderno e contextualizado.

// Cargo.toml: thiserror = "1", anyhow = "1"
use thiserror::Error;

#[derive(Error, Debug)]
enum MyError {
    #[error("IO error: {0}")]
    Io(#[from] std::io::Error),
    #[error("Custom error: {0}")]
    Custom(String),
}

fn main() -> anyhow::Result<()> {
    std::fs::File::open("arquivo_inexistente.txt")?;
    Ok(())
}

13 — Lazy Static Initialization

Inicialização preguiçosa de dados globais thread-safe.

// Cargo.toml: lazy_static = "1.4"
use lazy_static::lazy_static;
use std::collections::HashMap;

lazy_static! {
    static ref CONFIG: HashMap<String, String> = {
        let mut m = HashMap::new();
        m.insert("host".to_string(), "localhost".to_string());
        m
    };
}

fn main() {
    println!("Host: {}", CONFIG.get("host").unwrap());
}

14 — Advanced Pattern Matching + Guards

Pattern matching poderoso com guards.

enum Message {
    Quit,
    Move { x: i32, y: i32 },
    Write(String),
}

fn process(msg: Message) {
    match msg {
        Message::Quit => println!("Quit"),
        Message::Move { x, y } if x + y > 100 => println!("Movimento grande"),
        Message::Move { x, y } => println!("Move para ({}, {})", x, y),
        Message::Write(s) if s.len() > 10 => println!("Texto longo: {}", s),
        Message::Write(s) => println!("Texto: {}", s),
    }
}

fn main() {
    process(Message::Move { x: 60, y: 50 });
    process(Message::Write("Olá Rust!".to_string()));
}

15 — Custom Allocator (Global Allocator)

Controle fino sobre alocação de memória (útil em sistemas embarcados ou performance crítica).

use std::alloc::{GlobalAlloc, System, Layout};

struct MyAllocator;

unsafe impl GlobalAlloc for MyAllocator {
    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
        println!("Alocando {} bytes", layout.size());
        System.alloc(layout)
    }
    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
        System.dealloc(ptr, layout);
    }
}

#[global_allocator]
static ALLOCATOR: MyAllocator = MyAllocator;

fn main() {
    let v = vec![1; 1000];
    println!("Vetor de {} elementos criado", v.len());
}
