01 - Converte números decimais em binários

Transforma um inteiro armazenado no BD em sua representação binária (string).

-- 01 - Converte números decimais em binários
-- UTILIDADE: Transforma um inteiro armazenado no BD em sua representação binária (string).
-- CAMPO DE OPERAÇÃO: valor_decimal (INT)
CREATE TABLE IF NOT EXISTS numeros_exemplo (
    id INT PRIMARY KEY,
    valor_decimal INT NOT NULL
);
INSERT INTO numeros_exemplo (id, valor_decimal) VALUES (1, 10), (2, 255), (3, 42);

-- MySQL: BIN() | SQL Server: CONVERT(varchar, valor_decimal, 2)
SELECT id, valor_decimal, BIN(valor_decimal) AS representacao_binaria
FROM numeros_exemplo;
-- Altere 'valor_decimal' pelo nome da sua coluna inteira.

02 - Verifica se um bit específico está ativado

Filtra registros onde uma flag/permissão específica (ex: 4 = 2^2) está ligada.

-- 02 - Verifica se um bit específico está ativado
-- UTILIDADE: Filtra registros onde uma flag/permissão específica (ex: 4 = 2^2) está ligada.
-- CAMPO DE OPERAÇÃO: permissoes_mask (INT)
CREATE TABLE IF NOT EXISTS usuarios_permissao (
    id INT PRIMARY KEY,
    nome VARCHAR(50),
    permissoes_mask INT NOT NULL
);
INSERT INTO usuarios_permissao (id, nome, permissoes_mask) VALUES 
(1, 'Ana', 7), (2, 'Beto', 3), (3, 'Carlos', 12);

SELECT id, nome, permissoes_mask
FROM usuarios_permissao
WHERE (permissoes_mask & 4) = 4; -- Retorna Ana e Carlos. Troque '4' pela potência de 2 desejada.

03 - Ativa (liga) um bit específico

Adiciona uma permissão/status sem sobrescrever as flags existentes.

-- 03 - Ativa (liga) um bit específico
-- UTILIDADE: Adiciona uma permissão/status sem sobrescrever as flags existentes.
-- CAMPO DE OPERAÇÃO: flags_sistema (INT)
CREATE TABLE IF NOT EXISTS config_sistema (
    id INT PRIMARY KEY,
    modulo VARCHAR(50),
    flags_sistema INT DEFAULT 0
);
INSERT INTO config_sistema (id, modulo, flags_sistema) VALUES (1, 'Cache', 5); -- 0101

UPDATE config_sistema 
SET flags_sistema = flags_sistema | 32 -- 32 = 2^5 (liga o 6º bit)
WHERE id = 1;
-- Resultado: flags_sistema = 37 (100101). Altere '32' e 'id = 1' conforme necessidade.

04 - Desativa (desliga) um bit específico

Remove uma permissão usando máscara invertida (~), mantendo os outros bits intactos.

-- 04 - Desativa (desliga) um bit específico
-- UTILIDADE: Remove uma permissão usando máscara invertida (~), mantendo os outros bits intactos.
-- CAMPO DE OPERAÇÃO: nivel_acesso (INT)
CREATE TABLE IF NOT EXISTS contas_usuarios (
    id INT PRIMARY KEY,
    usuario VARCHAR(50),
    nivel_acesso INT DEFAULT 0
);
INSERT INTO contas_usuarios (id, usuario, nivel_acesso) VALUES (1, 'admin', 15); -- 1111

UPDATE contas_usuarios
SET nivel_acesso = nivel_acesso & ~8 -- ~8 = ...11110111. Zera apenas o 4º bit.
WHERE id = 1;
-- Resultado: nivel_acesso = 7 (0111). Troque '~8' pelo bit que deseja desligar.

05 - Inverte (toggle) um bit específico

Alterna o estado de um flag (0→1 ou 1→0) com XOR.

-- 05 - Inverte (toggle) um bit específico
-- UTILIDADE: Alterna o estado de um flag (0→1 ou 1→0) com XOR.
-- CAMPO DE OPERAÇÃO: status_flag (INT)
-- OBS: PostgreSQL usa # em vez de ^.
CREATE TABLE IF NOT EXISTS modulos_app (
    id INT PRIMARY KEY,
    nome_modulo VARCHAR(50),
    status_flag INT DEFAULT 0
);
INSERT INTO modulos_app (id, nome_modulo, status_flag) VALUES (1, 'Notificações', 3); -- 0011

UPDATE modulos_app
SET status_flag = status_flag ^ 2 -- XOR com 2 (0010) inverte apenas o 2º bit
WHERE id = 1;
-- 1ª execução: 3^2=1 | 2ª execução: 1^2=3. Altere '^ 2' pela flag de toggle.

06 - Combina múltiplas flags em um único valor

Gera um inteiro único a partir de várias permissões para INSERT/UPDATE.

-- 06 - Combina múltiplas flags em um único valor
-- UTILIDADE: Gera um inteiro único a partir de várias permissões para INSERT/UPDATE.
-- CAMPO DE OPERAÇÃO: mask_permissao (INT) na tabela de destino
CREATE TABLE IF NOT EXISTS perfis_acesso (
    id INT PRIMARY KEY,
    nome_perfil VARCHAR(50),
    mask_permissao INT
);

-- Template de INSERT combinando flags:
INSERT INTO perfis_acesso (id, nome_perfil, mask_permissao) 
VALUES (1, 'Editor', (1 | 2 | 8)); -- 1|2|8 = 11 (00001011)

-- Ou UPDATE:
UPDATE perfis_acesso 
SET mask_permissao = (4 | 16 | 32) 
WHERE id = 1;
-- Altere os números pelas potências de 2 que representam suas permissões.

07 - Verifica sobreposição de permissões entre dois conjuntos

Junta tabelas quando há AO MENOS UMA permissão em comum.

-- 07 - Verifica sobreposição de permissões entre dois conjuntos
-- UTILIDADE: Junta tabelas quando há AO MENOS UMA permissão em comum.
-- CAMPO DE OPERAÇÃO: u.permissoes (INT) e r.permissoes_necessarias (INT)
CREATE TABLE IF NOT EXISTS usuarios_acesso (id INT, nome VARCHAR(50), permissoes INT);
CREATE TABLE IF NOT EXISTS recursos_sistema (id INT, nome_recurso VARCHAR(50), permissoes_necessarias INT);

INSERT INTO usuarios_acesso VALUES (1, 'Maria', 6);
INSERT INTO recursos_sistema VALUES (10, 'Relatório', 4);

SELECT u.nome, r.nome_recurso
FROM usuarios_acesso u
JOIN recursos_sistema r 
  ON (u.permissoes & r.permissoes_necessarias) <> 0; -- <> 0 = há interseção de bits
-- Troque os campos após o ON pelos nomes das suas colunas de máscara.

08 - Extrai um intervalo de bits (máscara + deslocamento)

Decodifica campos compactados (ex: bits 5-8 guardam versão, 1-4 guardam tipo).

-- 08 - Extrai um intervalo de bits (máscara + deslocamento)
-- UTILIDADE: Decodifica campos compactados (ex: bits 5-8 guardam versão, 1-4 guardam tipo).
-- CAMPO DE OPERAÇÃO: codigo_compactado (INT)
CREATE TABLE IF NOT EXISTS produtos_sku (
    id INT PRIMARY KEY,
    sku VARCHAR(20),
    codigo_compactado INT
);
INSERT INTO produtos_sku (id, sku, codigo_compactado) VALUES (1, 'PROD-A', 173), (2, 'PROD-B', 93);

SELECT id, sku, 
       (codigo_compactado >> 4) & 15 AS versao_extraida -- >>4 desloca, &15 (1111) isola
FROM produtos_sku;
-- Ajuste '>> 4' (bits a pular) e '& 15' (máscara de quantos bits ler) conforme seu layout.

09 - Verifica par/ímpar de forma otimizada

Substitui MOD(coluna, 2) por bitwise. O LSB (bit 0) define paridade.

-- 09 - Verifica par/ímpar de forma otimizada
-- UTILIDADE: Substitui MOD(coluna, 2) por bitwise. O LSB (bit 0) define paridade.
-- CAMPO DE OPERAÇÃO: quantidade (INT)
CREATE TABLE IF NOT EXISTS itens_estoque (
    id INT PRIMARY KEY,
    produto VARCHAR(50),
    quantidade INT
);
INSERT INTO itens_estoque (id, produto, quantidade) VALUES (1, 'Parafuso', 100), (2, 'Porca', 75);

SELECT produto, quantidade,
       CASE WHEN (quantidade & 1) = 1 THEN 'Ímpar' ELSE 'Par' END AS tipo_paridade
FROM itens_estoque;
-- Altere 'quantidade' pelo nome da sua coluna numérica.

10 - Filtro de múltiplos status com máscara única

Busca registros que possuem QUALQUER um de vários status ativos.

-- 10 - Filtro de múltiplos status com máscara única
-- UTILIDADE: Busca registros que possuem QUALQUER um de vários status ativos.
-- CAMPO DE OPERAÇÃO: status_flags (INT)
CREATE TABLE IF NOT EXISTS tickets_suporte (
    id INT PRIMARY KEY,
    assunto VARCHAR(100),
    status_flags INT
);
INSERT INTO tickets_suporte (id, assunto, status_flags) VALUES 
(1, 'Bug crítico', 10), (2, 'Dúvida simples', 1), (3, 'Erro de rede', 8);

-- 10 = 1010₂. Retorna se o 2º OU o 4º bit estiverem ativos.
SELECT id, assunto, status_flags
FROM tickets_suporte
WHERE (status_flags & 10) <> 0;
-- Substitua '10' pela soma das potências de 2 que representam os status de interesse.
