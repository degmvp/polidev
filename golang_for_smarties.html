1. Fatorial Recursivo

Exemplo clássico de recursão: a função chama a si mesma até atingir o caso base.

package main

import "fmt"

func Fatorial(n int) int {
    if n == 0 {
        return 1
    }

    return n * Fatorial(n-1)
}

func main() {
    fmt.Println(Fatorial(5))
}

2. Soma de Fatias

Mostra o uso de slices e o laço range, uma das estruturas mais importantes em Go.

package main

import "fmt"

func Soma(nums []int) int {
    total := 0

    for _, num := range nums {
        total += num
    }

    return total
}

func main() {
    fmt.Println(Soma([]int{1, 2, 3, 4, 5}))
}

3. Encontrar o Maior Valor

Varre uma fatia e preserva o maior valor encontrado. A função trata fatia vazia com retorno zero.

package main

import "fmt"

func Max(nums []int) int {
    if len(nums) == 0 {
        return 0
    }

    max := nums[0]

    for _, num := range nums {
        if num > max {
            max = num
        }
    }

    return max
}

func main() {
    fmt.Println(Max([]int{1, 3, 5, 2, 4}))
}

4. Ordenação por Inserção

Algoritmo didático de ordenação, excelente para entender movimentação interna de elementos.

package main

import "fmt"

func InsertionSort(arr []int) {
    for i := 1; i < len(arr); i++ {
        key := arr[i]
        j := i - 1

        for j >= 0 && arr[j] > key {
            arr[j+1] = arr[j]
            j--
        }

        arr[j+1] = key
    }
}

func main() {
    nums := []int{5, 2, 9, 1, 5, 6}
    InsertionSort(nums)
    fmt.Println(nums)
}

5. Função de Filtro

Implementa uma ideia de programação funcional: receber uma função como critério.

package main

import "fmt"

func Filtrar(nums []int, cond func(int) bool) []int {
    var resultado []int

    for _, num := range nums {
        if cond(num) {
            resultado = append(resultado, num)
        }
    }

    return resultado
}

func main() {
    nums := []int{1, 2, 3, 4, 5}

    pares := Filtrar(nums, func(n int) bool {
        return n%2 == 0
    })

    fmt.Println(pares)
}

6. Função de Mapeamento

Aplica uma função a cada elemento de uma fatia e devolve uma nova fatia transformada.

package main

import "fmt"

func Map(nums []int, fn func(int) int) []int {
    resultado := make([]int, len(nums))

    for i, num := range nums {
        resultado[i] = fn(num)
    }

    return resultado
}

func main() {
    nums := []int{1, 2, 3}

    quadrados := Map(nums, func(n int) int {
        return n * n
    })

    fmt.Println(quadrados)
}

7. Calcular a Média

Reaproveita a função Soma e converte o resultado para float64.

package main

import "fmt"

func Soma(nums []int) int {
    total := 0

    for _, num := range nums {
        total += num
    }

    return total
}

func Media(nums []int) float64 {
    if len(nums) == 0 {
        return 0
    }

    total := Soma(nums)
    return float64(total) / float64(len(nums))
}

func main() {
    fmt.Println(Media([]int{1, 2, 3, 4, 5}))
}

8. Busca Binária

Busca eficiente em uma fatia previamente ordenada. Retorna o índice ou -1.

package main

import "fmt"

func BuscaBinaria(arr []int, x int) int {
    esquerda, direita := 0, len(arr)-1

    for esquerda <= direita {
        meio := esquerda + (direita-esquerda)/2

        if arr[meio] == x {
            return meio
        }

        if arr[meio] < x {
            esquerda = meio + 1
        } else {
            direita = meio - 1
        }
    }

    return -1
}

func main() {
    nums := []int{1, 2, 3, 4, 5}
    fmt.Println(BuscaBinaria(nums, 3))
}

9. Verificar Palíndromo

Compara caracteres das extremidades em direção ao centro da string.

package main

import "fmt"

func EhPalindromo(s string) bool {
    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-1-i] {
            return false
        }
    }

    return true
}

func main() {
    fmt.Println(EhPalindromo("radar"))
}

10. Gerar Fibonacci

Gera uma fatia com os primeiros termos da sequência de Fibonacci, protegendo os casos pequenos.

package main

import "fmt"

func Fibonacci(n int) []int {
    if n <= 0 {
        return []int{}
    }

    if n == 1 {
        return []int{0}
    }

    fib := make([]int, n)
    fib[0], fib[1] = 0, 1

    for i := 2; i < n; i++ {
        fib[i] = fib[i-1] + fib[i-2]
    }

    return fib
}

func main() {
    fmt.Println(Fibonacci(10))
}
