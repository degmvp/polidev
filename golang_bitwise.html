1. Verificar se um Número é Potência de Dois

Usa a técnica clássica n & (n - 1). Se o resultado for zero, existe apenas um bit ligado.

package main

import "fmt"

func EhPotenciaDeDois(n int) bool {
    return n > 0 && (n&(n-1)) == 0
}

func main() {
    fmt.Println(EhPotenciaDeDois(8))
    fmt.Println(EhPotenciaDeDois(10))
}

2. Contar Bits 1 em um Número

Conta quantos bits estão ligados deslocando o número para a direita.

package main

import "fmt"

func ContarBits1(n int) int {
    count := 0

    for n > 0 {
        count += n & 1
        n >>= 1
    }

    return count
}

func main() {
    fmt.Println(ContarBits1(7))
}

3. Inverter os Bits de um Número

O operador ^ em Go inverte todos os bits do número.

package main

import "fmt"

func InverterBits(n int) int {
    return ^n
}

func main() {
    fmt.Println(InverterBits(5))
}

4. Trocar Dois Números Usando XOR

Troca valores sem variável auxiliar usando XOR.

package main

import "fmt"

func Trocar(a, b *int) {
    *a = *a ^ *b
    *b = *a ^ *b
    *a = *a ^ *b
}

func main() {
    a, b := 5, 10
    Trocar(&a, &b)
    fmt.Println(a, b)
}

5. Adicionar Dois Números sem Operador de Adição

O XOR calcula a soma sem carry; o AND calcula o carry.

package main

import "fmt"

func Adicionar(a, b int) int {
    for b != 0 {
        carry := a & b
        a = a ^ b
        b = carry << 1
    }

    return a
}

func main() {
    fmt.Println(Adicionar(15, 32))
}

6. Multiplicar Dois Números Usando Bitwise

Usa deslocamento à esquerda para dobrar e à direita para reduzir o multiplicador.

package main

import "fmt"

func Multiplicar(a, b int) int {
    resultado := 0

    for b > 0 {
        if b&1 == 1 {
            resultado += a
        }

        a <<= 1
        b >>= 1
    }

    return resultado
}

func main() {
    fmt.Println(Multiplicar(5, 3))
}

7. Dividir Dois Números Usando Bitwise

Versão usando deslocamentos para montar o quociente bit a bit.

package main

import (
    "fmt"
    "strconv"
)

func Dividir(dividendo, divisor int) int {
    if divisor <= 0 {
        return 0
    }

    resultado := 0

    for i := strconv.IntSize - 1; i >= 0; i-- {
        if (divisor << i) <= dividendo {
            dividendo -= divisor << i
            resultado |= 1 << i
        }
    }

    return resultado
}

func main() {
    fmt.Println(Dividir(15, 3))
}

8. Obter o Máximo de Dois Números Usando Bitwise

Em Go, booleano não vira inteiro automaticamente. Por isso usamos uma função auxiliar.

package main

import "fmt"

func boolToInt(b bool) int {
    if b {
        return 1
    }
    return 0
}

func Max(a, b int) int {
    mask := -boolToInt(a < b)
    return a ^ ((a ^ b) & mask)
}

func main() {
    fmt.Println(Max(10, 20))
}

9. Verificar se um Número é Par ou Ímpar

Se o último bit for zero, o número é par.

package main

import "fmt"

func EhPar(n int) bool {
    return (n & 1) == 0
}

func main() {
    fmt.Println(EhPar(4))
    fmt.Println(EhPar(5))
}

10. Limitar um Número a um Intervalo

Versão corrigida para Go usando máscaras inteiras.

package main

import "fmt"

func boolToInt(b bool) int {
    if b {
        return 1
    }
    return 0
}

func Limitar(n, min, max int) int {
    maskMin := -boolToInt(n < min)
    maskMax := -boolToInt(n > max)

    return (n & ^maskMin & ^maskMax) |
           (min & maskMin) |
           (max & maskMax)
}

func main() {
    fmt.Println(Limitar(15, 10, 20))
    fmt.Println(Limitar(25, 10, 20))
}

Observação Polydev

Dois exemplos originais precisavam de ajuste para Go puro: -(a < b) e expressões como min & (n < min) não compilam em Go, porque Go não converte booleanos automaticamente para inteiros.
