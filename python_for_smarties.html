# ====================== 10 FUNÇÕES PYTHON FOR SMARTIES #2 ======================

from functools import cache, lru_cache, partial, reduce
from itertools import chain, tee, zip_longest, islice
import time
from contextlib import contextmanager
from collections import defaultdict


# 1 - Cache Infinito (Memoization)
@cache
def fibonacci(n: int) -> int:
    """Fibonacci ultra-rápido com cache infinito (LRU sem limite)"""
    return n if n <= 1 else fibonacci(n-1) + fibonacci(n-2)


# 2 - Context Manager para medir tempo
@contextmanager
def timer(name: str = "Operação"):
    """Context manager elegante para medir tempo de execução"""
    start = time.perf_counter()
    try:
        yield
    finally:
        end = time.perf_counter()
        print(f"⏱️  {name} executada em {end - start:.6f} segundos")


# 3 - Partial + Currying Simples
def power(base: int, exponent: int) -> int:
    return base ** exponent

square = partial(power, exponent=2)
cube = partial(power, exponent=3)


# 4 - Agrupar itens por chave (mais poderoso que groupby)
def group_by(items: list, key_func):
    """Agrupa lista de dicionários ou objetos por qualquer critério"""
    grouped = defaultdict(list)
    for item in items:
        grouped[key_func(item)].append(item)
    return dict(grouped)


# 5 - Flatten aninhado arbitrário
def flatten(nested_list):
    """Achata listas aninhadas em qualquer nível de profundidade"""
    for item in nested_list:
        if isinstance(item, list):
            yield from flatten(item)
        else:
            yield item


# 6 - Pipeline de funções (estilo functional)
def pipeline(value, *functions):
    """Aplica várias funções em sequência (pipe)"""
    return reduce(lambda v, f: f(v), functions, value)


# 7 - Gerador de janelas deslizantes (sliding window)
def sliding_window(iterable, size: int):
    """Gera janelas deslizantes de tamanho fixo"""
    iters = tee(iterable, size)
    for i, it in enumerate(iters):
        next(islice(it, i, None), None)   # avançar cada iterador
    return zip(*iters)


# 8 - Detectar duplicatas mantendo ordem
def unique_preserve_order(items):
    """Remove duplicatas mantendo a ordem de aparição (Python 3.7+)"""
    seen = set()
    return [x for x in items if not (x in seen or seen.add(x))]


# 9 - Memoization com TTL (expiração)
from functools import wraps
import time as _time

def ttl_cache(seconds: int = 60):
    """Decorator de cache com tempo de vida (TTL)"""
    def decorator(func):
        cache_dict = {}
        @wraps(func)
        def wrapper(*args, **kwargs):
            key = args + tuple(sorted(kwargs.items()))
            now = _time.time()
            if key in cache_dict:
                result, timestamp = cache_dict[key]
                if now - timestamp < seconds:
                    return result
            result = func(*args, **kwargs)
            cache_dict[key] = (result, now)
            return result
        return wrapper
    return decorator


# 10 - Operador de composição de funções
def compose(*functions):
    """Compor funções: compose(f, g, h)(x) = f(g(h(x)))"""
    def composed(x):
        for f in reversed(functions):
            x = f(x)
        return x
    return composed
